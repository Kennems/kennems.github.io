class Solution:
    def getCoprimes(self, nums: List[int], edges: List[List[int]]) -> List[int]:
        n = len(nums)
        g = [[]*n for _ in range(n)]
        ret = [-1]*n
        store = [[] for _ in range(51)]
        mem = [(-1, -1)]*(51) 

        for i in range(1, 51):
            for j in range(1, 51):
                if gcd(i, j)==1:
                    store[i].append(j)

        for u, v in edges:
            g[u].append(v)
            g[v].append(u)

        def dfs(u, last, level):
            ret[u] = max(mem[i] for i in store[nums[u]])[1]
            tmp = mem[nums[u]]
            mem[nums[u]] = (level, u)
            for son in g[u]:
                if son==last:
                    continue
                dfs(son, u, level+1)
            mem[nums[u]] = tmp

        dfs(0, -1, 0)

        return ret
