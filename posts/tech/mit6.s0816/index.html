void backtrace(void)
{
  uint64 p = r_fp();
  while (p != PGROUNDUP(p))
  {
    uint64 ra = *(uint64 *)(p - 8);
    uint64 fp = *(uint64 *)(p - 16);
    printf("%p\n", ra);
    p = fp;
  }
}
