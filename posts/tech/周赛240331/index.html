from sortedcontainers import SortedList
class Solution:
    def minimumDistance(self, points: List[List[int]]) -> int:
        xset, yset = SortedList(), SortedList()
        for x, y in points:
            xset.add(x+y)
            yset.add(x-y)
        ans = inf
        for x, y in points:
            xx = x+y
            yy = x-y 
            xset.remove(xx)
            yset.remove(yy)
            ans = min( ans, max(xset[-1]-xset[0], yset[-1]-yset[0]) )
            xset.add(xx)
            yset.add(yy)
        return ans
