class Solution:
    def findKthSmallest(self, coins: List[int], k: int) -> int:
        n = len(coins)
        ls = []
        for i in range(1, 1<<n):
            Lcm = 1
            for j, x in enumerate(coins):
                if i>>j & 1:
                    Lcm = lcm(Lcm, x)
            ls.append((1, Lcm)) if i.bit_count()&1 else ls.append((-1, Lcm))
        l, r = 1, int(5e10)
        while l<r:
            mid = l+r>>1
            cnt = 0
            for sign, val in ls:
                cnt += sign*(mid//val)
            if cnt>=k:
                r = mid
            else:
                l = mid+1
        return l

class Solution:
    def minimumValueSum(self, nums: List[int], andValues: List[int]) -> int:
        n = len(nums)
        m = len(andValues)

        @cache
        def dfs(i, j, k):
            if i==n:
                return 0 if j==m else inf
            if j==m:
                return inf
            k&=nums[i]
            if k<andValues[j]:
                return inf
            res = dfs(i+1, j, k)
            if k==andValues[j]:
                res = min(res, dfs(i+1, j+1, -1)+nums[i])
            return res
        ans = dfs(0, 0, -1)
        return ans if ans<inf else -1
