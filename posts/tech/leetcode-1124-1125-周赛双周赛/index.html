                heappush(sel, -heappop(hq))

class Solution:
    def maxCollectedFruits(self, fruits: List[List[int]]) -> int:
        n = len(fruits)

        # å®šä¹‰ä¸€ä¸ªé€’å½’å‡½æ•°ï¼Œè®¡ç®—ä» (i, j) åˆ°ç»ˆç‚¹çš„æœ€å¤§æ”¶é›†æ°´æœæ•°

            # æ£€æŸ¥åæ ‡æ˜¯å¦è¶Šç•Œï¼ˆé™åˆ¶åœ¨ç½‘æ ¼å†…ï¼‰
            if not (n - 1 - i <= j < n):
                return -inf
            # å¦‚æœåˆ°è¾¾èµ·ç‚¹ (0, x)ï¼Œç›´æ¥è¿”å›æˆ¿é—´ä¸­çš„æ°´æœæ•°é‡
            if i == 0:
                return fruits[i][j]

            return max(
                dfs(i - 1, j - 1),  # å·¦ä¸Š
                dfs(i - 1, j),      # ä¸Š
                dfs(i - 1, j + 1)   # å³ä¸Š
            ) + fruits[i][j]

        # ç»Ÿè®¡ç¬¬ä¸€æ¡è·¯å¾„ (0, 0) å‡ºå‘çš„å°æœ‹å‹æ”¶é›†çš„æ°´æœ

        # è®¡ç®—ç¬¬äºŒæ¡è·¯å¾„ï¼Œä»å³ä¸Šè§’ (0, n-1) å‡ºå‘
        ans += dfs(n - 2, n - 1)  # ä»å€’æ•°ç¬¬äºŒè¡Œå¾€ä¸Šé€’å½’è®¡ç®—
        dfs.cache_clear()

        # å¯¹ç½‘æ ¼è½¬ç½®ï¼Œç›¸å½“äºä¸»å¯¹è§’çº¿ç¿»è½¬ï¼Œç”¨äºè®¡ç®—ç¬¬ä¸‰æ¡è·¯å¾„

        ans += dfs(n - 2, n - 1)

class Solution:
    def minArraySum(self, nums: List[int], k: int, op1: int, op2: int) -> int:
        # åˆå§‹åŒ–å˜é‡
        n = len(nums)
        # åˆ›å»º DP è¡¨ fï¼Œå…¶ä¸­ f[i][j][x] è¡¨ç¤ºå¤„ç†å‰ i ä¸ªå…ƒç´ æ—¶ï¼Œä½¿ç”¨ j æ¬¡æ“ä½œ 1 å’Œ x æ¬¡æ“ä½œ 2 çš„æœ€å°å’Œ
        f = [[[float('inf')] * (op2 + 1) for _ in range(op1 + 1)] for _ in range(2)]
        f[0][0][0] = 0  # åˆå§‹çŠ¶æ€ï¼Œæœªæ“ä½œæ—¶å’Œä¸º 0

        # åŠ¨æ€è§„åˆ’è¿­ä»£
        for i in range(1, n + 1):  # éå†æ¯ä¸ªå…ƒç´ 
            cur, prev = i % 2, (i - 1) % 2  # å½“å‰å’Œå‰ä¸€çŠ¶æ€çš„ç´¢å¼•
            for j in range(op1 + 1):  # éå†æ“ä½œ 1 çš„ä½¿ç”¨æ¬¡æ•°
                for x in range(op2 + 1):  # éå†æ“ä½œ 2 çš„ä½¿ç”¨æ¬¡æ•°
                    # ä¸å¯¹ nums[i-1] æ“ä½œ
                    f[cur][j][x] = f[prev][j][x] + nums[i - 1]

                    # å¦‚æœå¯ä»¥ä½¿ç”¨æ“ä½œ 1
                    if j > 0:
                        f[cur][j][x] = min(
                            f[cur][j][x], 
                            f[prev][j - 1][x] + math.ceil(nums[i - 1] / 2)
                        )

                    # å¦‚æœå¯ä»¥ä½¿ç”¨æ“ä½œ 2 ä¸” nums[i-1] >= k
                    if x > 0 and nums[i - 1] >= k:
                        f[cur][j][x] = min(
                            f[cur][j][x], 
                            f[prev][j][x - 1] + max(nums[i - 1] - k, 0)
                        )

                    # å¦‚æœå¯ä»¥åŒæ—¶ä½¿ç”¨æ“ä½œ 1 å’Œæ“ä½œ 2
                    if j > 0 and x > 0:
                        reduced = math.ceil(nums[i - 1] / 2)  # æ“ä½œ 1 åçš„å€¼
                        if reduced >= k:  # æ“ä½œ 1 åä»å¯åº”ç”¨æ“ä½œ 2
                            f[cur][j][x] = min(
                                f[cur][j][x], 
                                f[prev][j - 1][x - 1] + max(reduced - k, 0)
                            )
                        if nums[i - 1] >= k:  # æ“ä½œ 2 åä»å¯åº”ç”¨æ“ä½œ 1
                            after_sub = nums[i - 1] - k
                            f[cur][j][x] = min(
                                f[cur][j][x], 
                                f[prev][j - 1][x - 1] + math.ceil(after_sub / 2)
                            )

        # è®¡ç®—æœ€ç»ˆç»“æœ
        res = float('inf')
        for j in range(op1 + 1):
            for x in range(op2 + 1):
                res = min(res, f[n % 2][j][x])  # å–æ‰€æœ‰å¯èƒ½çŠ¶æ€çš„æœ€å°å€¼

        return res

class Solution:
    def maximizeSumOfWeights(self, edges: List[List[int]], k: int) -> int:
        # å»ºç«‹é‚»æ¥è¡¨å­˜å‚¨æ ‘çš„ç»“æ„
        graph = defaultdict(list)
        for x, y, weight in edges:
            graph[x].append((y, weight))
            graph[y].append((x, weight))
        
        # æ·±åº¦ä¼˜å…ˆæœç´¢ï¼Œè¿”å› (ä¸é€‰å½“å‰èŠ‚ç‚¹çš„æœ€å¤§æƒé‡å’Œ, é€‰å½“å‰èŠ‚ç‚¹çš„æœ€å¤§æƒé‡å’Œ)
        def dfs(node: int, parent: int) -> Tuple[int, int]:

            not_choose = 0
            # `inc` å­˜å‚¨æ‰€æœ‰å­æ ‘ä¸­æƒé‡çš„å¢é‡å€¼ï¼ˆé€‰æ‹© vs ä¸é€‰æ‹©å­æ ‘ï¼‰
            increments = []

            # éå†å­èŠ‚ç‚¹
            for neighbor, weight in graph[node]:
                if neighbor == parent:
                    continue  # è·³è¿‡çˆ¶èŠ‚ç‚¹ï¼Œé¿å…å›ç¯
                nc, c = dfs(neighbor, node)  # å­èŠ‚ç‚¹é€’å½’è®¡ç®—
                not_choose += nc  # ç´¯åŠ å­èŠ‚ç‚¹ä¸é€‰æ—¶çš„æœ€å¤§æƒé‡å’Œ
                # å¦‚æœé€‰æ‹©å­èŠ‚ç‚¹å¯ä»¥å¢åŠ æƒé‡ï¼Œå°†å¢é‡å­˜å…¥ `increments`
                delta = c + weight - nc
                if delta > 0:
                    increments.append(delta)
            
            # æŒ‰å¢é‡ä»å¤§åˆ°å°æ’åºï¼Œå–å‰ k æˆ– k-1 ä¸ª
            increments.sort(reverse=True)
            choose_k = not_choose + sum(increments[:k])  # é€‰ k ä¸ªå¢é‡
            choose_k_minus_1 = not_choose + sum(increments[:k - 1])  # é€‰ k-1 ä¸ªå¢é‡
            
            return choose_k, choose_k_minus_1
        
        # é€’å½’ä»æ ¹èŠ‚ç‚¹å¼€å§‹ï¼Œè¿”å›æœ€å¤§å¯èƒ½çš„æƒé‡å’Œ
        return max(dfs(0, -1))
