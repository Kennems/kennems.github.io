            for j in range(n):

# Primç®—æ³•æ±‚è§£æœ€å°ç”Ÿæˆæ ‘çš„æƒé‡ä¹‹å’Œ
def prim(n, g):
    dist = [INF] * (n + 1)
    st = [False] * (n + 1)
    res = 0
    for i in range(n):
        t = -1
        for j in range(1, n + 1):
            if not st[j] and (t == -1 or dist[t] > dist[j]):
                t = j
        if i and dist[t] == INF:
            return INF
        if i:
            res += dist[t]
        st[t] = True
        for j in range(1, n + 1):
            dist[j] = min(dist[j], g[t][j])
    return res

def bfs(u, c):
    color[u] = c
    q = deque()
    q.append((u, c))

            elif color[nx]==col:
                return False
    return True
