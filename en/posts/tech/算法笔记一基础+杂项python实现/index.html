def quick_sort(q, l, r):
    if l>=r:
        return 
    i,j,x=l-1,r+1,q[(l+r)>>1]
    while i<j:
        i+=1
        while q[i]<x:
            i+=1
        j-=1
        while q[j]>x:
            j-=1
        if i<j:
            q[i], q[j] = q[j], q[i]
    quick_sort(q, l, j)
    quick_sort(q, j+1, r)

    merge_sort(q, mid+1, r)

def prefixSum2D(arr):
    n = len(arr)
    m = len(arr[0])
    prefixSum = [[0] * m for _ in range(n)]

    # è®¡ç®—ç¬¬ä¸€è¡Œçš„å‰ç¼€å’Œ
    prefixSum[0][0] = arr[0][0]
    for j in range(1, m):
        prefixSum[0][j] = prefixSum[0][j-1] + arr[0][j]

    # è®¡ç®—ç¬¬ä¸€åˆ—çš„å‰ç¼€å’Œ

    return prefixSum

def merge(segs):
    segs.sort()  # åŒºé—´å·¦ç«¯ç‚¹æ’åº
    res = []
    st, ed = -2e9, -2e9
    for seg in segs:
        if ed < seg[0]:
            if st != -2e9:
                res.append((st, ed))
            st, ed = seg[0], seg[1]
        else:
            ed = max(ed, seg[1])
    if st != -2e9:
        res.append((st, ed))
    segs[:] = res
